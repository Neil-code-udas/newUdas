select * from tb_material
DROP TABLE IF EXISTS tb_material;
CREATE TABLE tb_material (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '主键ID',
    material_name VARCHAR(100) NOT NULL COMMENT '物资名称（唯一）',
    unit VARCHAR(20) DEFAULT NULL COMMENT '单位：支/盒/本',
    price DECIMAL(16,2) DEFAULT 0 COMMENT '采购单价',
    stock DECIMAL(16,2) NOT NULL DEFAULT 0 COMMENT '当前实时库存',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_material_name (material_name) -- 名称唯一，禁止重复物资
) COMMENT '办公用品基础档案';

select * from tb_stock_record;
DROP TABLE IF EXISTS tb_stock_record;
CREATE TABLE tb_stock_record (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    material_id BIGINT NOT NULL COMMENT '关联物资ID',
    change_type VARCHAR(10) NOT NULL COMMENT 'IN入库 / OUT出库',
    change_num DECIMAL(16,2) NOT NULL COMMENT '变动数量',
    before_stock DECIMAL(16,2) NOT NULL COMMENT '变动前库存',
    after_stock DECIMAL(16,2) NOT NULL COMMENT '变动后库存',
    use_person VARCHAR(50) COMMENT '出库领用人，入库为空',
    remark VARCHAR(500) COMMENT '备注说明',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '操作时间'
) COMMENT '出入库台账流水';
